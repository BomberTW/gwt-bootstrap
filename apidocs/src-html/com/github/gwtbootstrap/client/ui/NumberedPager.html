027/**
028 * A pager for controlling a {@link HasRows} using {@link Pagination}.
029 * Just bind the NumberedPager using {@code NumberedPager.setDisplay} to some {@link HasRows} implementation like
030 * {@link CellTable} or {@link DataGrid}, and the pager is automatically generated fully working.
031 *
032 * <p>
033 * <h3>Example</h3>
034 *
035 * <pre>
036 *   <code>
037 *     <b:NumberedPager display="{table}"/>
038 *     <b:CellTable pageSize="10" ui:field="table" width="100%" />
039 *   </code>
040 * </pre>
041 *
042 * </p>
043 */

220    @Override
221    protected void onRangeOrRowCountChanged() {
222        int pageCount = super.getPageCount();
223        if (pageCount > 0) {
224
225                // shows all pages and rebuilt incrementally
226                if (maxPages < 0) {
227                        if (pagination.getWidgetCount() == 0) {
228                                // Lazy init for not displaying back and forward buttons unnecessarily
229                                initPagination();
230                        }
231                        int widgetCount = pagination.getWidgetCount();
232                        if (pageCount + 2 > widgetCount) {
233                                // If has there are *more* pages then links, then add the remaining links
234                                for (int i = widgetCount-1; i <= pageCount; i++) {
235                                        NavLink page = pagination.addPageLink(i);
236                                        page.addClickHandler(createPageClickHandler(i - 1));
237                                }
238                        } else if (pageCount + 2 < widgetCount) {
239                                // If has there are *less* pages then links, then remove the exceeding links
240                                for (int i = widgetCount - 2; i > pageCount; i--) {
241                                        pagination.remove(i);
242                                }
243                        }
244
245                        // shows a rotated fixed number of pages (determined by 'maxpages') and rebuilt from scratch
246                } else {
247                        // built it from scratch
248                        pagination.clear();
249
250                        // calculate offsets
251                        int currentPage = getPage();
252                        int pagesToShow = Math.min(getPageCount(), maxPages);
253                        int firstPageToShow = 1;
254
255                        if (maxPages > getPageCount())
256                                firstPageToShow = 1;
257                        else if (currentPage > getPageCount() - (maxPages / 2))
258                                firstPageToShow = getPageCount() - maxPages + 1;  
259                        else if(currentPage > maxPages / 2)
260                                firstPageToShow = currentPage - (maxPages / 2) + 1;                     
261
262                        // add prev
263                        final NavLink prev1 = new NavLink(prevText);
264                        prev1.addClickHandler(new ClickHandler() {
265                                @Override
266                                public void onClick(ClickEvent event) {
267                                        NumberedPager.this.previousPage();
268                                }
269                        });
270                        pagination.add(prev1);
271
272                        // add pages
273                        for(int i = 1; i < firstPageToShow; i++)                     // workaround. base component need all previous children, create fake (not visible)
274                                pagination.addPageLink(i).setVisible(false);
275                        for (int i = firstPageToShow; i < pagesToShow + firstPageToShow; i++) {
276                                NavLink page = pagination.addPageLink(i);
277                                page.addClickHandler(createPageClickHandler(i - 1));
278                        }
279
280                        // add next
281                        final NavLink next1 = new NavLink(nextText);
282                        next1.addClickHandler(new ClickHandler() {
283                                @Override
284                                public void onClick(ClickEvent event) {
285                                        NumberedPager.this.nextPage();
286                                }
287                        });
288                        pagination.add(next1);
289
290                        prev1.setDisabled(!super.hasPreviousPage());
291                        next1.setDisabled(!super.hasNextPage());
292                }
293
294                // Set the actual page link as disabled
295                updateButtonsState();
296        } else {
297                if (pagination.getWidgetCount() > 0) {
298                        pagination.clear();
299                }
300        }
301    }
302
303    private ClickHandler createPageClickHandler(final int page) {
304        return new ClickHandler() {
305            @Override
306            public void onClick(ClickEvent event) {
307                NumberedPager.this.setPage(page);
308            }
309        };
310    }
311
312    private void updateButtonsState() {
313        // Set all numbered buttons as enabled
314        for (int i = 1; i < pagination.getWidgetCount()-1; i++) {
315            NavLink navLink = (NavLink) pagination.getWidget(i);
316            navLink.setDisabled(false);
317            navLink.setActive(false);
318        }
319        // Set the button of the current page as disable
320        NavLink navLink = (NavLink) pagination.getWidget(super.getPage() + 1);
321        navLink.setDisabled(true);
322        navLink.setActive(true);
323        // Update the state of previous and next buttons
324        updatePreviousAndNextButtons();
325    }
326
327    private void initPagination() {
328        prev.addClickHandler(new ClickHandler() {
329            @Override
330            public void onClick(ClickEvent event) {
331                NumberedPager.this.previousPage();
332            }
333        });
334        pagination.add(prev);
335
336        NavLink page = pagination.addPageLink(1);
337        page.addClickHandler(new ClickHandler() {
338            @Override
339            public void onClick(ClickEvent event) {
340                NumberedPager.this.setPage(0);
341            }
342        });
343        page.setDisabled(true);
344
345        next.addClickHandler(new ClickHandler() {
346            @Override
347            public void onClick(ClickEvent event) {
348                NumberedPager.this.nextPage();
349            }
350        });
351        pagination.add(next);
352    }
353
354    private void updatePreviousAndNextButtons() {
355        prev.setDisabled(!super.hasPreviousPage());
356        next.setDisabled(!super.hasNextPage());
357    }
