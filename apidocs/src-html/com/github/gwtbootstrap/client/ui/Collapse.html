024/**
025 * Markup widget of Collapse
026 * <p>
027 * It's a markup widget (decorator widget).
028 * it's can exchange child to Collapsible widget.
029 * 
030 * It's need trigger, You have 2 ways to create trigger.<br/>
031 * <ol>
032 *   <li> Using {@link CollapseTrigger}. </li>
033 *   <li> Calling {@link #toggle()}. </li>
034 * </ol>
035 * </p>
036 * 
037 * <p>
038 * <h3>UiBinder Usage:</h3>
039 * </p>
040 * <pre>
041 * {@code
042 * <b:Collapse b:id="toggle1" defaultOpen="true">
043 *   <!-- it can be added any widget, but accept one wdiget.
044 *   <b:Label>aaa</b:Label>
045 * </b:Collapse>
046 * }
047 * </pre>
048 * 
049 * @since 2.2.1.0
050 * @author ohashi keisuke
051 * @see Accordion
052 * @see Collapse
053 * @see CollapseTrigger
054 * @see <a href="http://twitter.github.com/bootstrap/javascript.html#collapse">Twitter Bootstrap document</a>
055 * 
056 *
057 */
058public class Collapse extends MarkupWidget implements HasVisibility, HasVisibleHandlers {

104    public void setDefaultOpen(boolean dafaultOpen) {
105        this.dafaultOpen = dafaultOpen;
106        
107        if(widget != null && !widget.isAttached()) {
108            widget.setStyleName(Constants.IN , dafaultOpen);
109        }
110        
111    }

113    /**
114     * {@inheritDoc}
115     */
116    @Override
117    public Widget asWidget() {
118        
119        if(widget != null) {
120            Scheduler.get().scheduleDeferred(new ScheduledCommand() {
121                
122                @Override
123                public void execute() {
124                    if(!isExistTrigger()){
125                        reconfigure();
126                    } else {
127                        setHandlerFunctions(widget.getElement());
128                    }
129                }
130            });
131        }
132        
133        return getWidget();
134    }
135

137     * {@inheritDoc}

139    @Override
140    public void setWidget(Widget w) {
141        super.setWidget(w);
142        
143        if(widget != null) {
144            widget.addStyleName(Constants.COLLAPSE);
145        }
146        
147    }

221    /**
222     * Change visibility
223     * @param visibilityChange call method
224     */
225    protected void changeVisibility(VisibilityChange visibilityChange) {
226        
227        if(widget == null) return;
228        
229        changeVisibility(widget.getElement() , visibilityChange.get());
230    }
231
232    protected void changeVisibility(VisibilityChange visibilityChange, boolean autoTriggered) {
233
234        if(widget == null) return;
235
236        changeVisibility(widget.getElement() , visibilityChange.get(), autoTriggered);
237    }

255    /**
256     * re configure setting
257     */
258    public void reconfigure() {
259        
260        if(widget == null) return;
261        
262        setDefaultOpen(dafaultOpen);
263        
264        removeDataIfExists(widget.getElement());
265        
266        setHandlerFunctions(widget.getElement());
267        
268        configure(widget.getElement(), parent, toggle);
269    }

397    /**
398     * This method is called immediately when the widget's {@link #hide()}
399     * method is executed.
400     */
401    protected void onHide(Event e) {
402        widget.fireEvent(new HideEvent(e, getAutoTriggered(e)));
403    }
404
405    /**
406     * This method is called once the widget is completely hidden.
407     */
408    protected void onHidden(Event e) {
409        widget.fireEvent(new HiddenEvent(e, getAutoTriggered(e)));
410    }
411
412    /**
413     * This method is called immediately when the widget's {@link #show()}
414     * method is executed.
415     */
416    protected void onShow(Event e) {
417        widget.fireEvent(new ShowEvent(e, getAutoTriggered(e)));
418    }
419
420    /**
421     * This method is called once the widget is completely shown.
422     */
423    protected void onShown(Event e) {
424        widget.fireEvent(new ShownEvent(e, getAutoTriggered(e)));
