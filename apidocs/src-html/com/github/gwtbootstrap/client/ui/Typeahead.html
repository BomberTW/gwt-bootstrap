030    private int displayItems = 8;
031
032    private int minLength = 1;
033
034    private final SuggestOracle oracle;

036
037    private UpdaterCallback updaterCallback;
038    private HighlighterCallback highlighterCallback;
039
040    /**
041     * Constructor for {@link Typeahead}. Creates a {@link MultiWordSuggestOracle} to use with this
042     */
043    public Typeahead() {
044        this(new MultiWordSuggestOracle());
045    }
046
047    /**
048     * Constructor for {@link Typeahead}.
049     *
050     * @param oracle the oracle for this <code>Typeahead</code>
051     */
052    public Typeahead(SuggestOracle oracle) {
053        this.oracle = oracle;
054        this.updaterCallback = createDefaultUpdaterCallback();
055        this.highlighterCallback = createDefaultHighlighterCallback();
056    }
057
058    private UpdaterCallback createDefaultUpdaterCallback() {
059        return new UpdaterCallback() {
060            @Override
061            public String onSelection(Suggestion selectedSuggestion) {
062                return selectedSuggestion.getReplacementString();
063            }
064        };
065    }
066
067    private HighlighterCallback createDefaultHighlighterCallback() {
068        return new HighlighterCallback() {
069            @Override
070            public String highlight(String item) {
071                return item;
072            }
073        };
074    }
075
076    /**
077     * {@inheritDoc}
078     */
079    @Override
080    public void setWidget(Widget w) {
081
082        if (!(w instanceof TextBoxBase || w instanceof com.google.gwt.user.client.ui.TextBoxBase)) {
083            throw new IllegalArgumentException("Typeahead should be set TextBoxBase childs");
084        }
085
086        super.setWidget(w);
087    }
088
089    @Override
090    public Widget asWidget() {
091
092        if (widget != null) {
093            Scheduler.get().scheduleDeferred(new ScheduledCommand() {
094                @Override
095                public void execute() {
096                    reconfigure();
097                }
098            });
099
100        }
101
102        return super.asWidget();
103    }
104
105    /**
106     * reconfigure setting.
107     */
108    public void reconfigure() {
109
110        if (widget == null) {
111            return;
112        }
113
114        removeDataIfExists(widget.getElement());
115
116        configure(widget.getElement(), displayItems, minLength);
117    }
118

121     *

127

130     *
131     * @param displayItems the max number of items to display in the dropdown

135    }
136
137    /**
138     * Get the minimum character length needed before triggering autocomplete suggestions.
139     *
140     * @return the minimum character length needed before triggering autocomplete suggestions

143        return minLength;
144    }
145
146    /**
147     * Set the minimum character length needed before triggering autocomplete suggestions.
148     *
149     * @param minLength The minimum character length needed before triggering autocomplete suggestions.

151    public void setMinLength(int minLength) {
152        this.minLength = minLength;
153    }
154
155    public void setUpdaterCallback(UpdaterCallback updaterCallback) {
156        this.updaterCallback = updaterCallback;
157    }
158
159    public void setHighlighterCallback(
160            HighlighterCallback highlighterCallback) {
161        this.highlighterCallback = highlighterCallback;
162    }
163
164    /**
165     * Get suggest oracle
166     *
167     * @return oracle
168     */
169    public SuggestOracle getSuggestOracle() {
170        return oracle;
171    }
172
173    private void query(String query, final JavaScriptObject process) {
174        Callback callback = new Callback() {
175            @Override
176            public void onSuggestionsReady(Request request, Response response) {
177                callback(process, response);
178            }
179        };
180
181        if (query != null && !query.isEmpty()) {
182            oracle.requestSuggestions(new Request(query, displayItems), callback);
183        } else {
184            oracle.requestDefaultSuggestions(new Request(), callback);
185        }
186    }
187
188    private void callback(final JavaScriptObject process, Response response) {
189        suggestions = response.getSuggestions();
190
191        JsArrayString jsArrayString = JavaScriptObject.createArray().cast();
192
193        for (Suggestion suggestion : suggestions) {
194            jsArrayString.push(suggestion.getDisplayString());
195        }
196        callProcess(jsArrayString, process);
197    }
198
199    private String updater(String item) {
200        for (Suggestion suggestion : suggestions) {
201            if (suggestion.getDisplayString().equals(item)) {
202                return this.updaterCallback.onSelection(suggestion);
203            }
204        }
205
206        return item;
207    }
208
209    private String highlighter(String item) {
210        return this.highlighterCallback.highlight(item);
211    }
