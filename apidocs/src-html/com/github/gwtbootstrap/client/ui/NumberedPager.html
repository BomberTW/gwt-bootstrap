034/**
035 * A pager for controlling a {@link HasRows} using {@link Pagination}.
036 * Just bind the NumberedPager using {@code NumberedPager.setDisplay} to some {@link HasRows} implementation like
037 * {@link CellTable} or {@link DataGrid}, and the pager is automatically generated fully working.
038 * <p/>
039 * <p>
040 * <h3>Example</h3>
041 * <p/>
042 * <pre>
043 *   <code>
044 *     <b:NumberedPager display="{table}"/>
045 *     <b:CellTable pageSize="10" ui:field="table" width="100%" />
046 *   </code>
047 * </pre>
048 * <p/>
049 * </p>
050 */
051public class NumberedPager extends AbstractPager implements HasStyle, IsResponsive, HasId {

189    @Override
190    protected void onRangeOrRowCountChanged() {
191        final HasRows display = super.getDisplay();
192        final int pageSize = super.getPageSize();
193        final int calculatedPage = display.getVisibleRange().getStart() / pageSize;
194        final int pageCount = super.getPageCount();
195
196        // Workaround GWT misbehavior:
197        // Calculating wrong page when dynamically changing pageSize to a lower value and start index is between two pages.
198        // This logic aims to correct the start index.
199        if (pageCount > 0 && (calculatedPage != super.getPage())) {
200            display.setVisibleRange(pageSize * calculatedPage, pageSize);
201        } else {
202            // Proceed with normal paging logic
203            if (pageCount > 0) {
204                if (pagination.getWidgetCount() == 0) {
205                    // Lazy init for not displaying back and forward buttons unnecessarily
206                    initPagination();
207                }
208
209                int widgetCount = pagination.getWidgetCount();
210                if (pageCount + 2 > widgetCount) {
211                    // If there are *more* pages then links, then add the remaining links
212                    for (int i = widgetCount - 1; i <= pageCount; i++) {
213                        NavLink page = pagination.addPageLink(i);
214                        page.addClickHandler(createPageClickHandler(i - 1));
215                    }
216                } else if (pageCount + 2 < widgetCount) {
217                    // If there are *less* pages then links, then remove the exceeding links
218                    for (int i = widgetCount - 2; i > pageCount; i--) {
219                        pagination.remove(i);
220                    }
221                }
222
223                // Set the actual page link as disabled
224                updateButtonsState();
225            } else {
226                if (pagination.getWidgetCount() > 0) {
227                    resetPagination();
228                }
229            }
230        }
231    }
232
233    private void initPagination() {
234        if (previousCustomStyleName != null) previousLink.addStyleName(previousCustomStyleName);
235        handlerRegistrationList.add(previousLink.addClickHandler(new ClickHandler() {
236            @Override
237            public void onClick(ClickEvent event) {
238                NumberedPager.this.previousPage();
239            }
240        }));
241        pagination.add(previousLink);
242
243        NavLink page = pagination.addPageLink(1);
244        page.addClickHandler(new ClickHandler() {
245            @Override
246            public void onClick(ClickEvent event) {
247                NumberedPager.this.setPage(0);
248            }
249        });
250        page.setDisabled(true);
251
252        if (nextCustomStyleName != null) nextLink.addStyleName(nextCustomStyleName);
253        handlerRegistrationList.add(nextLink.addClickHandler(new ClickHandler() {
254            @Override
255            public void onClick(ClickEvent event) {
256                NumberedPager.this.nextPage();
257            }
258        }));
259        pagination.add(nextLink);
260    }
261
262    private void updateButtonsState() {
263        final int pageCount = getPageCount();
264        final int currentPage = getPage();
265
266        // If visiblePages is set, then treat circular exhibition
267        if (visiblePages > 0) {
268            // Calculate offsets
269            final int maxPages = visiblePages < pageCount ? visiblePages : pageCount;
270            final int pagesToShow = Math.min(pageCount, maxPages);
271            int firstVisibleIndex = 1;
272
273            if (currentPage >= pageCount - (maxPages / 2)) {
274                firstVisibleIndex = pageCount - maxPages + 1;
275            } else if (currentPage > maxPages / 2) {
276                firstVisibleIndex = currentPage - (maxPages / 2) + 1;
277            }
278
279            final int firstNotVisibleIndex = pagesToShow + firstVisibleIndex;
280
281            // Set out of range pages as invisible (before initial threshold)
282            for (int i = 1; i < firstVisibleIndex; i++) {
283                pagination.getWidget(i).setVisible(false);
284            } // (after final threshold)
285            for (int i = firstNotVisibleIndex; i < pagination.getWidgetCount() - 1; i++) {
286                pagination.getWidget(i).setVisible(false);
287            }
288
289            // Set in range pages as visible
290            for (int i = firstVisibleIndex; i < firstNotVisibleIndex; i++) {
291                pagination.getWidget(i).setVisible(true);
292            }
293        }
294
295        // Set all numbered buttons as enabled
296        for (int i = 1; i < pagination.getWidgetCount() - 1; i++) {
297            NavLink navLink = (NavLink) pagination.getWidget(i);
298            navLink.setDisabled(false);
299            navLink.setActive(false);
300        }
301
302        // Set the button of the current page as disable
303        NavLink navLink = (NavLink) pagination.getWidget(currentPage + 1);
304        navLink.setDisabled(true);
305        navLink.setActive(true);
306
307        // Update the state of previous and next buttons
308        updatePreviousAndNextButtons();
309    }
310
311    private void updatePreviousAndNextButtons() {
312        previousLink.setDisabled(!super.hasPreviousPage());
313        nextLink.setDisabled(!super.hasNextPage());
314    }
315
316    private ClickHandler createPageClickHandler(final int page) {
317        return new ClickHandler() {
318            @Override
319            public void onClick(ClickEvent event) {
320                NumberedPager.this.setPage(page);
321            }
322        };
323    }
324
325    private void resetPagination() {
326        // Deregister click handlers
327        for (HandlerRegistration handlerRegistration : handlerRegistrationList) {
328            handlerRegistration.removeHandler();
329        }
330        pagination.clear();
331        if (previousCustomStyleName != null) previousLink.removeStyleName(previousCustomStyleName);
332        if (nextCustomStyleName != null) nextLink.removeStyleName(nextCustomStyleName);
333    }
