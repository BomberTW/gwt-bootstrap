059/**
060 * CheckBox widgets.
061 * <p>
062 * Re-design for Bootstrap.
063 * </p>
064 *
065 * <p>CheckBox Usage:</p>
066 * <pre>{@code
067 * <b:CheckBox formValue="remember-me" text="Remember me"/>
068 * }</pre>
069 * <p>Above code renders to...</p>
070 * <pre>{@code
071 *  <label class="checkbox pull-left" for="gwt-uid-n">
072 *      <input id="gwt-uid-n" type="checkbox" value="remember-me" tabindex="0"></input>
073 *      <span>Remember me</span>
074 *  </label>}</pre>
075 *  ...where n is any number generated by GWT.
076 * @since 2.0.4.0
077 * @author ohashi keisuke
078 */
079public class CheckBox extends ButtonBase implements HasName, HasValue<Boolean>, HasWordWrap, HasDirectionalSafeHtml, HasDirectionEstimator, IsEditor<LeafValueEditor<Boolean>>, IsSearchQuery, HasId, IsResponsive, HasStyle{

206        protected CheckBox(Element elem) {
207                super(DOM.createLabel());
208                
209                assert elem.hasAttribute("type") : "The elem should has type attributes";
210
211                //TODO 2012/05/06 ohashi keisuke. ugly code......
212                if(Constants.CHECKBOX.toLowerCase().equals(elem.getAttribute("type").toLowerCase())) {
213                        this.setStyleName(Constants.CHECKBOX);
214                } else if(Constants.RADIO.toLowerCase().equals(elem.getAttribute("type").toLowerCase())){
215                        this.setStyleName(Constants.RADIO);
216                }
217
218                inputElem = InputElement.as(elem);
219                spanElem = Document.get().createSpanElement();
220                
221                getElement().appendChild(inputElem);
222                getElement().appendChild(spanElem);
223
224                String uid = DOM.createUniqueId();
225                inputElem.setPropertyString("id", uid);
226                asLabel().setHtmlFor(uid);
227                directionalTextHelper = new DirectionalTextHelper(spanElem, false);
228                setTabIndex(0);
229        }

231        public HandlerRegistration addValueChangeHandler(ValueChangeHandler<Boolean> handler) {
232                // Is this the first value change handler? If so, time to add handlers
233                if (!valueChangeHandlerInitialized) {
234                        ensureDomEventHandlers();
235                        valueChangeHandlerInitialized = true;
236                }
237                return addHandler(handler, ValueChangeEvent.getType());
238        }

449        /**
450         * Checks or unchecks the check box, firing {@link ValueChangeEvent} if
451         * appropriate.
452         * <p>
453         * Note that this <em>does not</em> set the value property of the checkbox
454         * input element wrapped by this widget. For access to that property, see
455         * {@link #setFormValue(String)}
456         * 
457         * @param value
458         *            true to check, false to uncheck; null value implies false
459         * @param fireEvents
460         *            If true, and value has changed, fire a
461         *            {@link ValueChangeEvent}
462         */
463        public void setValue(Boolean value, boolean fireEvents) {
464                if (value == null) {
465                        value = Boolean.FALSE;
466                }
467
468                Boolean oldValue = getValue();
469                inputElem.setChecked(value);
470                inputElem.setDefaultChecked(value);
471                if (value.equals(oldValue)) {
472                        return;
473                }
474                if (fireEvents) {
475                        ValueChangeEvent.fire(this, value);
476                }
477        }

485        // Unlike other widgets the CheckBox sinks on its inputElement, not
486        // its wrapper
487        @Override
488        public void sinkEvents(int eventBitsToAdd) {
489                if (isOrWasAttached()) {
490                        Event.sinkEvents(inputElem, eventBitsToAdd | Event.getEventsSunk(inputElem));
491                } else {
492                        super.sinkEvents(eventBitsToAdd);
493                }
494        }

496        protected void ensureDomEventHandlers() {
497                addClickHandler(new ClickHandler() {
498
499                        public void onClick(ClickEvent event) {
500                                // Checkboxes always toggle their value, no need to compare
501                                // with old value. Radio buttons are not so lucky, see
502                                // overrides in RadioButton
503                                ValueChangeEvent.fire(CheckBox.this, getValue());
504                        }
505                });
506        }

535        /**
536         * This method is called when a widget is detached from the browser's
537         * document. Overridden because of IE bug that throws away checked state and
538         * in order to clear the event listener off of the <code>inputElem</code>.
539         */
540        @Override
541        protected void onUnload() {
542                // Clear out the inputElem's event listener (breaking the circular
543                // reference between it and the widget).
544                setEventListener(asOld(inputElem), null);
545                setValue(getValue());
546        }

548        /**
549         * Replace the current input element with a new one. Preserves all state
550         * except for the name property, for nasty reasons related to radio button
551         * grouping. (See implementation of {@link RadioButton#setName}.)
552         * 
553         * @param elem
554         *            the new input element
555         */
556        protected void replaceInputElement(Element elem) {
557                InputElement newInputElem = InputElement.as(elem);
558                // Collect information we need to set
559                int tabIndex = getTabIndex();
560                boolean checked = getValue();
561                boolean enabled = isEnabled();
562                String formValue = getFormValue();
563                String uid = inputElem.getId();
564                String accessKey = inputElem.getAccessKey();
565                int sunkEvents = Event.getEventsSunk(inputElem);
566
567                // Clear out the old input element
568                setEventListener(asOld(inputElem), null);
569
570                getElement().replaceChild(newInputElem, inputElem);
571
572                // Sink events on the new element
573                Event.sinkEvents(elem, Event.getEventsSunk(inputElem));
574                Event.sinkEvents(inputElem, 0);
575                inputElem = newInputElem;
576
577                // Setup the new element
578                Event.sinkEvents(inputElem, sunkEvents);
579                inputElem.setId(uid);
580                if (!"".equals(accessKey)) {
581                        inputElem.setAccessKey(accessKey);
582                }
583                setTabIndex(tabIndex);
584                setValue(checked);
585                setEnabled(enabled);
586                setFormValue(formValue);
587
588                // Set the event listener
589                if (isAttached()) {
590                        setEventListener(asOld(inputElem), this);
591                }
592        }
593
594        private Element asOld(com.google.gwt.dom.client.Element elem) {
595                Element oldSchool = elem.cast();
596                return oldSchool;
597        }

601        }
