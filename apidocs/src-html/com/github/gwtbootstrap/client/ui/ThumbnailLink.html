001package com.github.gwtbootstrap.client.ui;
002
003import com.github.gwtbootstrap.client.ui.base.HasSize;
004import com.github.gwtbootstrap.client.ui.base.SizeHelper;
005import com.github.gwtbootstrap.client.ui.constants.Constants;
006
007/**
008 * The thumbnail be wrapped by Anchor.
009 * 
010 * @since 2.2.1.0
011 * @author ohashi keisuke
012 * @see Thumbnails
013 * @see Thumbnail
014 * @see <a href="http://getbootstrap.com/2.3.2/components.html#thumbnails">Twitter Bootstrap Documentation</a>
015 */
016public class ThumbnailLink extends NavWidget implements HasSize {
017
018    /**
019     * Create an empty Thumbnail
020     */
021    public ThumbnailLink() {
022        super();
023        getAnchor().setStyleName(Constants.THUMBNAIL);
024    }
025
026    /**
027     * {@inheritDoc}
028     */
029    @Override
030    public void setSize(int size) {
031        SizeHelper.setSize(this, size);
032    }
033    
034}
